import spacy

# Load English tokenizer
nlp = spacy.load("en_core_web_sm")

# Sample text
text = """
Natural Language Processing (NLP) is a fascinating field of Artificial Intelligence.
It deals with the interaction between computers and humans using natural language.
Python provides many tools for text processing. Let's explore some of them!
"""
# --- Paragraph Tokenizer (based on newlines) ---
paragraphs = [p.strip() for p in text.strip().split('\n') if p.strip()]
print("ðŸ”¸ Paragraphs:\n")
for i, para in enumerate(paragraphs, 1):
    print(f"Paragraph {i}: {para}\n")

# Process text using spaCy
doc = nlp(text)

# --- Sentence Tokenizer ---
print("ðŸ”¸ Sentences:\n")
for i, sent in enumerate(doc.sents, 1):
    print(f"Sentence {i}: {sent.text}")

# --- Word Tokenizer ---
print("\nðŸ”¸ Word Tokens:\n")
word_tokens = [token.text for token in doc if not token.is_space]
print(word_tokens)
